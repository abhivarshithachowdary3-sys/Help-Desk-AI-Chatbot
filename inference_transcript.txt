# ── Load intent classifier if available ──────────────────────────
CKPT_DIR = os.path.join(os.path.dirname(os.path.abspath(__file__)), "..", "checkpoints")
_clf_bundle = None
def _load_clf():
    global _clf_bundle
    p = os.path.join(CKPT_DIR, "intent_classifier.pkl")
    if os.path.exists(p) and _clf_bundle is None:
        with open(p, "rb") as f:
            _clf_bundle = pickle.load(f)
    return _clf_bundle

def classify_intent(text):
    bundle = _load_clf()
    if not bundle:
        return "unknown", 1.0
    import numpy as np
    vec = bundle["vectorizer"].transform([text])
    probs = bundle["classifier"].predict_proba(vec)[0]
    idx = int(np.argmax(probs))
    conf = float(probs[idx])
    intent = bundle["classifier"].classes_[idx]
    return (intent, conf) if conf >= 0.35 else ("low_confidence", conf)

# ── Helpers ───────────────────────────────────────────────────────
def _now(): return datetime.datetime.now().strftime("%H:%M")
def _tokens(t): return max(1, len(t)//4)
def _md(text):
    import html
    t=html.escape(text)
    t=re.sub(r"\*\*(.*?)\*\*",r"<strong>\1</strong>",t)
    t=re.sub(r"\*(.*?)\*",r"<em>\1</em>",t)
    t=re.sub(r"`(.*?)`",r"<code>\1</code>",t)
    return t.replace("\n","<br>")
def export_json(chat):
    return json.dumps({"title":chat["title"],"persona":chat["persona"],
        "messages":[{"role":m["role"],"content":m["content"],"time":m["ts"]}
                    for m in chat["messages"]]},indent=2)
def export_txt(chat):
    lines=[f"── {chat['title']} | {chat['persona']} ──\n"]
    for m in chat["messages"]:
        lines.append(f"[{m['ts']}] {'You' if m['role']=='user' else 'Bot'}: {m['content']}\n")
    return "\n".join(lines)

# ── Session state ─────────────────────────────────────────────────
def init():
    for k,v in dict(conversations={},active_chat_id=None,api_key="",
                    selected_persona="🎧 HelpDesk Agent",show_tokens=True,
                    temperature=0.7,max_tokens=1024,file_content=None,
                    file_name=None,total_tokens=0,show_intents=True).items():
        if k not in st.session_state: st.session_state[k]=v

def new_chat(persona=None):
    cid=str(uuid.uuid4())[:8]
    p=persona or st.session_state.selected_persona
    st.session_state.conversations[cid]={
        "title":f"Chat #{len(st.session_state.conversations)+1}",
        "messages":[],"persona":p,"created":_now(),"tokens":0}
    st.session_state.active_chat_id=cid
    st.session_state.conversations[cid]["messages"].append(
        {"role":"assistant","content":PERSONAS[p]["greeting"],"ts":_now(),"tokens":0,"intent":""})
    return cid

def get_active():
    cid=st.session_state.active_chat_id
    return st.session_state.conversations.get(cid)

# ── Groq API ──────────────────────────────────────────────────────
def get_client():
    key=st.session_state.api_key or os.environ.get("GROQ_API_KEY","")
    return Groq(api_key=key) if key else None

def stream_response(client, chat, user_msg, file_content=None):
    system=PERSONAS[chat["persona"]]["system"]
    msgs=[{"role":"system","content":system}]
    for m in chat["messages"]:
        if m["role"] in ("user","assistant"):
            msgs.append({"role":m["role"],"content":m["content"]})
    content=user_msg
    if file_content:
        content=f"{user_msg}\n\n---\n**File:**\n```\n{file_content[:4000]}\n```"
    msgs.append({"role":"user","content":content})

    box=st.empty(); full=""
    try:
        stream=client.chat.completions.create(
            model="llama-3.3-70b-versatile",messages=msgs,
            temperature=st.session_state.temperature,
            max_tokens=st.session_state.max_tokens,stream=True)
        for chunk in stream:
            delta=chunk.choices[0].delta.content
            if delta:
                full+=delta
                box.markdown(f'<div class="bubble bot">{_md(full)} ▌</div>',unsafe_allow_html=True)
        box.empty()
    except Exception as e:
        full=f"⚠️ Error: {e}"; box.empty()
    tok=_tokens(full)
    chat["tokens"]=chat.get("tokens",0)+tok
    st.session_state.total_tokens+=tok
    return full,tok

            st.metric("Tokens used",f"{st.session_state.total_tokens:,}")

# ── Render messages ───────────────────────────────────────────────
def render_msgs(chat):
    for m in chat["messages"]:
        role=m["role"]; content=m["content"]
        ts=m.get("ts",""); tok=m.get("tokens",0)
        intent=m.get("intent","")
        badge=f'<span style="font-size:.7rem;color:#666;">{ts}</span>'
        if st.session_state.show_tokens and tok:
            badge+=f'<span style="background:#3f3d56;color:#a0a0c0;font-size:.68rem;padding:2px 8px;border-radius:20px;margin-left:6px">{tok} tok</span>'
        if st.session_state.show_intents and intent:
            badge+=f'<span class="intent-badge">🏷 {intent}</span>'
        if role=="user":
            st.markdown(f"""<div class="bubble-row user">
              <div><div class="bubble user">{_md(content)}</div><div class="ts">{badge}</div></div>
              <div class="avatar user">🧑</div></div>""",unsafe_allow_html=True)
        else:
            st.markdown(f"""<div class="bubble-row bot">
              <div class="avatar bot">🤖</div>
              <div><div class="bubble bot">{_md(content)}</div><div class="ts">{badge}</div></div>
            </div>""",unsafe_allow_html=True)

    if not st.session_state.active_chat_id:
        st.session_state.active_chat_id=list(st.session_state.conversations.keys())[0]
    chat=get_active()
    if not chat: st.info("Start a new chat from the sidebar."); return

    pk=chat["persona"]

    with c1:
        st.markdown(f"""<div class="main-header">
          <span>{pk.split()[0]}</span>

          <div style="color:#ccc;font-size:.85rem;">{PERSONAS[pk]['description']} · {chat['created']}</div>
          </div></div>""",unsafe_allow_html=True)
    with c2:
        if st.button("🔄 New"): new_chat(pk); st.rerun()

    with st.expander("📥 Export"):
        c1,c2=st.columns(2)
        with c1: st.download_button("⬇ JSON",export_json(chat),
                    file_name=f"helpdesk_{chat['title'][:20]}.json",
                    mime="application/json",use_container_width=True)

                    mime="text/plain",use_container_width=True)

    render_msgs(chat)

    if len(chat["messages"])<=2:

        if sugs:
            st.markdown("<div style='margin:8px 0 4px;color:#888;font-size:.8rem;'>Quick prompts:</div>",unsafe_allow_html=True)
            cols=st.columns(len(sugs))
            for i,s in enumerate(sugs):
                with cols[i]:
                    if st.button(s,key=f"s_{i}",use_container_width=True):
                        st.session_state["pending_text"]=s; st.rerun()

    with st.expander("📎 Attach file"):
        up=st.file_uploader("Include in context",type=["txt","py","md","json","csv"],key="fu")
        if up:
            try:
                st.session_state.file_content=up.read().decode("utf-8",errors="replace")
                st.session_state.file_name=up.name
                st.success(f"✓ {up.name} ({len(st.session_state.file_content):,} chars)")
            except Exception as e: st.error(str(e))
        if st.session_state.file_content:
            st.markdown(f'<div style="background:#2d2d3e;border-left:3px solid #6c63ff;padding:8px 14px;border-radius:8px;font-size:.82rem;color:#b0b0d0">📄 {st.session_state.file_name}</div>',unsafe_allow_html=True)
            if st.button("❌ Remove"): st.session_state.file_content=None; st.rerun()

    st.markdown("---")
    with st.form("chat_form",clear_on_submit=True):
        pre=st.session_state.pop("pending_text","") if "pending_text" in st.session_state else ""
        user_input=st.text_area("Message",value=pre,
            placeholder=f"Ask {pk.split()[1]} anything…",
            height=90,label_visibility="collapsed",key="mi")
        c1,c2,c3=st.columns([4,1,1])
        with c1: send=st.form_submit_button("Send ➤",use_container_width=True,type="primary")
        with c2: clear=st.form_submit_button("🗑 Clear",use_container_width=True)
        with c3: regen=st.form_submit_button("🔄 Regen",use_container_width=True)

    if clear:
        chat["messages"]=[{"role":"assistant","content":PERSONAS[pk]["greeting"],"ts":_now(),"tokens":0,"intent":""}]
        chat["tokens"]=0; st.rerun()
    if regen:
        msgs=chat["messages"]
        if msgs and msgs[-1]["role"]=="assistant" and len(msgs)>1:
            chat["messages"]=msgs[:-1]; st.rerun()

    if send and user_input.strip():
        client=get_client()
        if not client:
            st.error("⚠️ Enter your Groq API key in the sidebar!")
            st.info("Get FREE key: https://console.groq.com"); st.stop()

        # Intent classification — routes query + detects fallback need
        intent, confidence = classify_intent(user_input)
        if intent=="low_confidence":
            st.markdown(f'<div class="fallback-banner">⚠️ Low confidence ({confidence:.0%}) — escalating to human agent fallback</div>',unsafe_allow_html=True)

        chat["messages"].append({"role":"user","content":user_input.strip(),

            chat["title"]=user_input[:35]+("…" if len(user_input)>35 else "")

        with st.spinner(""):
            st.markdown("""<div class="bubble-row bot">
              <div class="avatar bot">🤖</div>
              <div class="thinking"><div class="dot"></div><div class="dot"></div><div class="dot"></div></div>
            </div>""",unsafe_allow_html=True)

                file_content=st.session_state.file_content)

        chat["messages"].append({"role":"assistant","content":reply,
            "ts":_now(),"tokens":tok,"intent":""})
        st.session_state.file_content=None; st.session_state.file_name=None
        st.rerun()
